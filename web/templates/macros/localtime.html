{#- Render a datetime as a element that JS converts to browser local time. Usage: {% from "macros/localtime.html" import localtime %} {{ localtime(device.first_seen) }} {{ localtime(device.last_seen, fmt='long') }} Formats: 'short' = YYYY-MM-DD HH:MM, 'long' = YYYY-MM-DD HH:MM:SS -#} {% macro localtime(dt, fmt='short', fallback='—') -%} {%- if dt -%} {{ dt.strftime('%Y-%m-%d %H:%M:%S') if fmt == 'long' else dt.strftime('%Y-%m-%d %H:%M') }} {%- else -%} {{ fallback }} {%- endif -%} {%- endmacro %}
